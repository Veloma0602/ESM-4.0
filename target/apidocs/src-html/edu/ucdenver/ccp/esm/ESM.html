012public class ESM {
013        DirectedGraph<Vertex, Edge> subgraph = null;

017        private double similarityThreshold = 0.8; // Define a threshold for similarity

019        public ESM(DirectedGraph<Vertex, Edge> subgraph, DirectedGraph<Vertex, Edge> graph, double similarityThreshold) {
020                if (subgraph == null || subgraph.getVertexCount() == 0) {
021                        throw new IllegalArgumentException("Subgraph cannot be null or empty.");
022                }
023                if (graph == null || graph.getVertexCount() == 0) {
024                        throw new IllegalArgumentException("Graph cannot be null or empty.");
025                }
026
027                this.graph = graph;
028                this.subgraph = subgraph;
029                this.similarityThreshold = similarityThreshold;
030                subgraphStartNode = getRandomStartNode(new ArrayList<>(subgraph.getVertices()));
031                graphStartNodes = new ArrayList<>(graph.getVertices());
032        }

034        private Vertex getRandomStartNode(List<Vertex> subgraphNodes) {
035                if (subgraphNodes.isEmpty()) {
036                        throw new IllegalArgumentException("subgraphNodes list cannot be empty.");
037                }
038                Random random = new Random();
039                int randomNumber = random.nextInt(subgraphNodes.size());
040                return subgraphNodes.get(randomNumber);
041        }

051        public List<Map<Vertex, Vertex>> getSubgraphMatchingMatches() {
052                List<Map<Vertex, Vertex>> matches = new ArrayList<>();
053                if (!isSubgraphSmaller()) {
054                        System.err.println("The size of the subgraph: " +
055                                        subgraph.getVertexCount() + " is bigger than the size of the graph " +
056                                        graph.getVertexCount() + ". Please check.");
057                        return matches;
058                }
059
060                for (Vertex graphStartNode : graphStartNodes) {
061                        Map<Vertex, Vertex> subgraphToGraph = new HashMap<>();
062                        Map<Vertex, Vertex> graphToSubgraph = new HashMap<>();
063                        List<Map<Vertex, Vertex>> total = new ArrayList<>();
064                        List<Vertex> toMatch = Arrays.asList(subgraphStartNode, graphStartNode);
065
066                        if (matchNodeForAllMatches(toMatch, subgraphToGraph, graphToSubgraph, subgraph, graph, total)) {
067                                for (Map<Vertex, Vertex> m : total) {
068                                        boolean flag = true;
069                                        for (Map<Vertex, Vertex> n : matches) {
070                                                if (equalMaps(m, n))
071                                                        flag = false;
072                                        }
073                                        if (flag) matches.add(m);
074                                }
075                        }
076                }
077                return matches;
078        }

080        private boolean equalMaps(Map<Vertex, Vertex> m1, Map<Vertex, Vertex> m2) {
081                if (m1.size() != m2.size())
082                        return false;
083                for (Vertex key : m1.keySet())
084                        if (!m1.get(key).equals(m2.get(key)))
085                                return false;
086                return true;
087        }

089        private boolean matchNodeForAllMatches(List<Vertex> toMatchs, Map<Vertex, Vertex> subgraphToGraphs, Map<Vertex, Vertex> graphToSubgraphs,
090                                                                                   DirectedGraph<Vertex, Edge> subgraph, DirectedGraph<Vertex, Edge> graph, List<Map<Vertex, Vertex>> total) {
091                List<Vertex> toMatch = new ArrayList<>(toMatchs);
092                Map<Vertex, Vertex> subgraphToGraph = new HashMap<>(subgraphToGraphs);
093                Map<Vertex, Vertex> graphToSubgraph = new HashMap<>(graphToSubgraphs);
094
095                while (!toMatch.isEmpty()) {
096                        Vertex noder = toMatch.remove(0);
097                        Vertex nodes = toMatch.remove(0);
098
099                        if (subgraphToGraph.containsKey(noder) && !graphToSubgraph.containsKey(nodes))
100                                return false;
101                        if (!subgraphToGraph.containsKey(noder) && graphToSubgraph.containsKey(nodes))
102                                return false;
103                        if (subgraphToGraph.containsKey(noder) && graphToSubgraph.containsKey(nodes)) {
104                                if (subgraphToGraph.get(noder).equals(nodes) && graphToSubgraph.get(nodes).equals(noder)) {
105                                        // Do nothing
106                                } else {
107                                        return false;
108                                }
109                        }
110
111                        if (!matchNodeContent(noder, nodes))
112                                return false;
113
114                        subgraphToGraph.put(noder, nodes);
115                        graphToSubgraph.put(nodes, noder);
116
117                        for (Edge e : subgraph.getOutEdges(noder)) {
118                                Vertex r = e.getDependent();
119                                List<Vertex> candidateNodes = new ArrayList<>();
120                                for (Edge s : graph.getOutEdges(nodes)) {
121                                        if (e.calculateCosineSimilarity(s) >= similarityThreshold)
122                                                candidateNodes.add(s.getDependent());
123                                }
124                                boolean flag = false;
125                                boolean terminate = false;
126                                for (Vertex s : candidateNodes) {
127                                        if (subgraphToGraph.containsKey(r) && !graphToSubgraph.containsKey(s))
128                                                continue;
129                                        if (!subgraphToGraph.containsKey(r) && graphToSubgraph.containsKey(s))
130                                                continue;
131                                        if (subgraphToGraph.containsKey(r) && graphToSubgraph.containsKey(s)) {
132                                                if (subgraphToGraph.get(r).equals(s) && graphToSubgraph.get(s).equals(r)) {
133                                                        terminate = true;
134                                                        break;
135                                                } else {
136                                                        continue;
137                                                }
138                                        }
139                                        List<Vertex> toMatchTemp = new ArrayList<>(toMatch);
140                                        toMatchTemp.add(noder);
141                                        toMatchTemp.add(nodes);
142                                        toMatchTemp.add(r);
143                                        toMatchTemp.add(s);
144                                        if (matchNodeForAllMatches(toMatchTemp, subgraphToGraph, graphToSubgraph, subgraph, graph, total))
145                                                flag = true;
146                                }
147                                if (terminate) continue;
148                                if (flag) return true;
149
150                                return false;
151                        }
152
153                        for (Edge e : subgraph.getInEdges(noder)) {
154                                Vertex r = e.getGovernor();
155                                List<Vertex> candidateNodes = new ArrayList<>();
156                                for (Edge s : graph.getInEdges(nodes)) {
157                                        if (e.calculateCosineSimilarity(s) >= similarityThreshold)
158                                                candidateNodes.add(s.getGovernor());
159                                }
160                                boolean flag = false;
161                                boolean terminate = false;
162                                for (Vertex s : candidateNodes) {
163                                        if (subgraphToGraph.containsKey(r) && !graphToSubgraph.containsKey(s))
164                                                continue;
165                                        if (!subgraphToGraph.containsKey(r) && graphToSubgraph.containsKey(s))
166                                                continue;
167                                        if (subgraphToGraph.containsKey(r) && graphToSubgraph.containsKey(s)) {
168                                                if (subgraphToGraph.get(r).equals(s) && graphToSubgraph.get(s).equals(r)) {
169                                                        terminate = true;
170                                                        break;
171                                                } else {
172                                                        continue;
173                                                }
174                                        }
175                                        List<Vertex> toMatchTemp = new ArrayList<>(toMatch);
176                                        toMatchTemp.add(noder);
177                                        toMatchTemp.add(nodes);
178                                        toMatchTemp.add(r);
179                                        toMatchTemp.add(s);
180                                        if (matchNodeForAllMatches(toMatchTemp, subgraphToGraph, graphToSubgraph, subgraph, graph, total))
181                                                flag = true;
182                                }
183                                if (terminate) continue;
184                                if (flag) return true;
185
186                                return false;
187                        }
188                }
189
190                total.add(subgraphToGraph);
191                return true;
192        }

194        public boolean isSubgraphIsomorphism() {
195                boolean isSubgraphIsomorphism = false;
196                if (!isSubgraphSmaller()) {
197                        System.err.println("The size of the subgraph: " +
198                                        subgraph.getVertexCount() + " is bigger the size of the graph " +
199                                        graph.getVertexCount() + ". Please check.");
200                        return isSubgraphIsomorphism;
201                }
202                for (int i = 0; i < graphStartNodes.size(); i++) {
203                        Map<Vertex, Vertex> subgraphToGraph = new HashMap<>();
204                        Map<Vertex, Vertex> graphToSubgraph = new HashMap<>();
205                        List<Vertex> toMatch = Arrays.asList(subgraphStartNode, graphStartNodes.get(i));
206
207                        if (matchNodeForSingleMatch(toMatch, subgraphToGraph, graphToSubgraph, subgraph, graph)) {
208                                isSubgraphIsomorphism = true;
209                                break;
210                        }
211                }
212                return isSubgraphIsomorphism;
213        }

215        public boolean isGraphIsomorphism() {
216                boolean isGraphIsomorphism = false;
217                boolean isSubgraphIsomorphicToGraph = false;
218                boolean isgraphIsomorphicToSubgraph = false;
219
220                if (!isGraphSizeSame()) {
221                        System.err.println("The size of the subgraph: " +
222                                        subgraph.getVertexCount() + " is not same as the size of the graph " +
223                                        graph.getVertexCount() + ". Please check.");
224                        return isGraphIsomorphism;
225                }
226                for (int i = 0; i < graphStartNodes.size(); i++) {
227                        Map<Vertex, Vertex> subgraphToGraph = new HashMap<>();
228                        Map<Vertex, Vertex> graphToSubgraph = new HashMap<>();
229                        List<Vertex> toMatch = Arrays.asList(subgraphStartNode, graphStartNodes.get(i));
230                        if (matchNodeForSingleMatch(toMatch, subgraphToGraph, graphToSubgraph, subgraph, graph)) {
231                                isSubgraphIsomorphicToGraph = true;
232                                break;
233                        }
234                }
235
236                subgraphStartNode = getRandomStartNode(new ArrayList<>(graph.getVertices()));
237                graphStartNodes = new ArrayList<>(subgraph.getVertices());
238                for (int i = 0; i < graphStartNodes.size(); i++) {
239                        Map<Vertex, Vertex> subgraphToGraph = new HashMap<>();
240                        Map<Vertex, Vertex> graphToSubgraph = new HashMap<>();
241
242                        List<Vertex> toMatch = Arrays.asList(subgraphStartNode, graphStartNodes.get(i));
243                        if (matchNodeForSingleMatch(toMatch, subgraphToGraph, graphToSubgraph, graph, subgraph)) {
244                                isgraphIsomorphicToSubgraph = true;
245                                break;
246                        }
247                }
248
249                if (isSubgraphIsomorphicToGraph && isgraphIsomorphicToSubgraph)
250                        isGraphIsomorphism = true;
251
252                subgraphStartNode = getRandomStartNode(new ArrayList<>(subgraph.getVertices()));
253                graphStartNodes = new ArrayList<>(graph.getVertices());
254
255                return isGraphIsomorphism;
256        }

258        private boolean matchNodeForSingleMatch(List<Vertex> toMatchs, Map<Vertex, Vertex> subgraphToGraphs, Map<Vertex, Vertex> graphToSubgraphs,
259                                                                                        DirectedGraph<Vertex, Edge> subgraph, DirectedGraph<Vertex, Edge> graph) {
260                List<Vertex> toMatch = new ArrayList<>(toMatchs);
261                Map<Vertex, Vertex> subgraphToGraph = new HashMap<>(subgraphToGraphs);
262                Map<Vertex, Vertex> graphToSubgraph = new HashMap<>(graphToSubgraphs);
263
264                while (!toMatch.isEmpty()) {
265                        Vertex noder = toMatch.remove(0);
266                        Vertex nodes = toMatch.remove(0);
267
268                        if (subgraphToGraph.containsKey(noder) && !graphToSubgraph.containsKey(nodes))
269                                return false;
270                        if (!subgraphToGraph.containsKey(noder) && graphToSubgraph.containsKey(nodes))
271                                return false;
272                        if (subgraphToGraph.containsKey(noder) && graphToSubgraph.containsKey(nodes)) {
273                                if (subgraphToGraph.get(noder).equals(nodes) && graphToSubgraph.get(nodes).equals(noder)) {
274                                        // Do nothing
275                                } else {
276                                        return false;
277                                }
278                        }
279
280                        if (!matchNodeContent(noder, nodes))
281                                return false;
282
283                        subgraphToGraph.put(noder, nodes);
284                        graphToSubgraph.put(nodes, noder);
285
286                        for (Edge e : subgraph.getOutEdges(noder)) {
287                                Vertex r = e.getDependent();
288                                List<Vertex> candidateNodes = new ArrayList<>();
289                                for (Edge s : graph.getOutEdges(nodes)) {
290                                        if (e.calculateCosineSimilarity(s) >= similarityThreshold)
291                                                candidateNodes.add(s.getDependent());
292                                }
293                                boolean terminate = false;
294                                for (Vertex s : candidateNodes) {
295                                        if (subgraphToGraph.containsKey(r) && !graphToSubgraph.containsKey(s))
296                                                continue;
297                                        if (!subgraphToGraph.containsKey(r) && graphToSubgraph.containsKey(s))
298                                                continue;
299                                        if (subgraphToGraph.containsKey(r) && graphToSubgraph.containsKey(s)) {
300                                                if (subgraphToGraph.get(r).equals(s) && graphToSubgraph.get(s).equals(r)) {
301                                                        terminate = true;
302                                                        break;
303                                                } else {
304                                                        continue;
305                                                }
306                                        }
307                                        List<Vertex> toMatchTemp = new ArrayList<>(toMatch);
308                                        toMatchTemp.add(noder);
309                                        toMatchTemp.add(nodes);
310                                        toMatchTemp.add(r);
311                                        toMatchTemp.add(s);
312                                        if (matchNodeForSingleMatch(toMatchTemp, subgraphToGraph, graphToSubgraph, subgraph, graph)) {
313                                                subgraphToGraphs.putAll(subgraphToGraph);
314                                                graphToSubgraphs.putAll(graphToSubgraph);
315                                                return true;
316                                        }
317                                }
318                                if (terminate) continue;
319
320                                return false;
321                        }
322
323                        for (Edge e : subgraph.getInEdges(noder)) {
324                                Vertex r = e.getGovernor();
325                                List<Vertex> candidateNodes = new ArrayList<>();
326                                for (Edge s : graph.getInEdges(nodes)) {
327                                        if (e.calculateCosineSimilarity(s) >= similarityThreshold)
328                                                candidateNodes.add(s.getGovernor());
329                                }
330                                boolean terminate = false;
331                                for (Vertex s : candidateNodes) {
332                                        if (subgraphToGraph.containsKey(r) && !graphToSubgraph.containsKey(s))
333                                                continue;
334                                        if (!subgraphToGraph.containsKey(r) && graphToSubgraph.containsKey(s))
335                                                continue;
336                                        if (subgraphToGraph.containsKey(r) && graphToSubgraph.containsKey(s)) {
337                                                if (subgraphToGraph.get(r).equals(s) && graphToSubgraph.get(s).equals(r)) {
338                                                        terminate = true;
339                                                        break;
340                                                } else {
341                                                        continue;
342                                                }
343                                        }
344                                        List<Vertex> toMatchTemp = new ArrayList<>(toMatch);
345                                        toMatchTemp.add(noder);
346                                        toMatchTemp.add(nodes);
347                                        toMatchTemp.add(r);
348                                        toMatchTemp.add(s);
349                                        if (matchNodeForSingleMatch(toMatchTemp, subgraphToGraph, graphToSubgraph, subgraph, graph)) {
350                                                subgraphToGraphs.putAll(subgraphToGraph);
351                                                graphToSubgraphs.putAll(graphToSubgraph);
352                                                return true;
353                                        }
354                                }
355                                if (terminate) continue;
356
357                                return false;
358                        }
359                }
360
361                subgraphToGraphs.putAll(subgraphToGraph);
362                graphToSubgraphs.putAll(graphToSubgraph);
363                return true;
364        }

366        private boolean matchNodeContent(Vertex noder, Vertex nodes) {
367                return noder.calculateCosineSimilarity(nodes) >= similarityThreshold;
368        }

370        private Boolean isSubgraphSmaller() {
371                return subgraph.getVertexCount() <= graph.getVertexCount();
372        }
373
374        private Boolean isGraphSizeSame() {
375                return subgraph.getVertexCount() == graph.getVertexCount();
